import pandas as pd
import numpy as np

# 创建示例数据
df = pd.DataFrame({
    'price': [10, 20, 30, 40, 50],
    'quantity': [2, 3, 1, 4, 2]
})

# 不好的做法 - 使用循环
total = []
for index, row in df.iterrows():
    total.append(row['price'] * row['quantity'])
df['total'] = total

# 好的做法 - 使用向量化操作
df['total'] = df['price'] * df['quantity']

# 更复杂的计算示例
df['discounted_price'] = df['price'] * np.where(df['quantity'] > 2, 0.9, 1.0)

# 定义复杂函数
def calculate_shipping_cost(row):
    base_cost = 5.0
    if row['weight'] > 10:
        base_cost += (row['weight'] - 10) * 0.5
    if row['distance'] > 100:
        base_cost += (row['distance'] - 100) * 0.1
    return base_cost

# 应用函数
df['shipping_cost'] = df.apply(calculate_shipping_cost, axis=1)

# 使用lambda函数进行简单转换
df['price_category'] = df['price'].apply(lambda x: '高' if x > 30 else '中' if x > 15 else '低')

    'sales': ['sum', 'mean', 'count'],

    return x.max() - x.min()

custom_agg = sales_data.groupby('region').agg({
    'sales': [sum, sales_range, lambda x: x.std()]
})

# 使用transform进行分组计算
sales_data['region_avg'] = sales_data.groupby('region')['sales'].transform('mean')
sales_data['performance'] = sales_data['sales'] / sales_data['region_avg']

# 创建包含异常值的数据
np.random.seed(42)
normal_data = np.random.normal(100, 15, 100)
outliers = np.array([200, 250, 10, 5])
data_with_outliers = np.concatenate([normal_data, outliers])
df_outliers = pd.DataFrame({'values': data_with_outliers})

# 使用Z-score识别异常值
from scipy import stats
df_outliers['z_score'] = np.abs(stats.zscore(df_outliers['values']))
outliers_zscore = df_outliers[df_outliers['z_score'] > 3]

# 使用IQR方法识别异常值
Q1 = df_outliers['values'].quantile(0.25)
Q3 = df_outliers['values'].quantile(0.75)
IQR = Q3 - Q1
lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR
outliers_iqr = df_outliers[
    (df_outliers['values'] < lower_bound) | 
    (df_outliers['values'] > upper_bound)
]

# 处理异常值
# 方法1: 删除
df_cleaned = df_outliers[
    (df_outliers['values'] >= lower_bound) & 
    (df_outliers['values'] <= upper_bound)
]

# 方法2: 替换为边界值
df_capped = df_outliers.copy()
df_capped['values'] = df_capped['values'].clip(lower_bound, upper_bound)

# 方法3: 替换为均值或中位数
median_value = df_outliers['values'].median()
df_replaced = df_outliers.copy()
df_replaced.loc[
    (df_replaced['values'] < lower_bound) | 
    (df_replaced['values'] > upper_bound),
    'values'
] = median_value

df_large = pd.DataFrame({
    'id': range(1000000),
    'category': ['A', 'B', 'C'] * 333333 + ['A'],
    'value': np.random.rand(1000000),
    'date': pd.date_range('2020-01-01', periods=1000000, freq='H')
})

# 检查内存使用
print("原始内存使用:")
print(df_large.memory_usage(deep=True))

# 优化数据类型
df_optimized = df_large.copy()
df_optimized['id'] = df_optimized['id'].astype('int32')  # 从int64改为int32
df_optimized['category'] = df_optimized['category'].astype('category')  # 改为category类型
df_optimized['value'] = df_optimized['value'].astype('float32')  # 从float64改为float32

print("\n优化后内存使用:")
print(df_optimized.memory_usage(deep=True))

# 自动优化数据类型的函数
def optimize_dtypes(df):
    for col in df.columns:
        col_type = df[col].dtype
        
        if col_type != 'object':
            c_min = df[col].min()
            c_max = df[col].max()
            
            if str(col_type)[:3] == 'int':
                if c_min > np.iinfo(np.int8).min and c_max < np.iinfo(np.int8).max:
                    df[col] = df[col].astype(np.int8)
                elif c_min > np.iinfo(np.int16).min and c_max < np.iinfo(np.int16).max:
                    df[col] = df[col].astype(np.int16)
                elif c_min > np.iinfo(np.int32).min and c_max < np.iinfo(np.int32).max:
                    df[col] = df[col].astype(np.int32)
            else:
                if c_min > np.finfo(np.float16).min and c_max < np.finfo(np.float16).max:
                    df[col] = df[col].astype(np.float16)
                elif c_min > np.finfo(np.float32).min and c_max < np.finfo(np.float32).max:
                    df[col] = df[col].astype(np.float32)
        else:
            df[col] = df[col].astype('category')
    
    return df

df_auto_optimized = optimize_dtypes(df_large.copy())

from sklearn.preprocessing import StandardScaler, LabelEncoder, OneHotEncoder
from sklearn.model_selection import train_test_split

# 创建示例数据
ml_data = pd.DataFrame({
    'age': [25, 30, 35, 40, 45, 50, 55, 60],
    'income': [30000, 40000, 50000, 60000, 70000, 80000, 90000, 100000],
    'education': ['高中', '本科', '本科', '硕士', '硕士', '博士', '博士', '博士'],
    'city': ['北京', '上海', '北京', '上海', '广州', '深圳', '北京', '上海'],
    'target': [0, 0, 1, 1, 1, 1, 1, 1]
})

# 处理分类变量
le = LabelEncoder()
ml_data['education_encoded'] = le.fit_transform(ml_data['education'])

# 独热编码
ohe = OneHotEncoder(sparse=False)
city_encoded = ohe.fit_transform(ml_data[['city']])
city_df = pd.DataFrame(
    city_encoded, 
    columns=ohe.get_feature_names_out(['city'])
)
ml_data = pd.concat([ml_data, city_df], axis=1)

# 特征缩放
scaler = StandardScaler()
numeric_features = ['age', 'income']
ml_data[numeric_features] = scaler.fit_transform(ml_data[numeric_features])

# 特征工程

            plt.tight_layout()

    # 目标变量分析
    if target_column and target_column in df.columns:
        print("\n" + "=" * 50)
        print("目标变量分析")
        print("=" * 50)
        print(df[target_column].describe())

        df[target_column].hist(bins=30)
        plt.title(f'{target_column} 分布')
        plt.xlabel(target_column)
        plt.ylabel('频次')
        plt.show()

from numba import jit

# 普通Python函数
def normal_sum(arr):
    total = 0
    for i in range(len(arr)):
        total += arr[i] * arr[i]
    return total

# 使用Numba优化的函数
@jit(nopython=True)
def numba_sum(arr):
    total = 0
    for i in range(len(arr)):
        total += arr[i] * arr[i]
    return total

# 性能比较
import time

large_array = np.random.rand(1000000)

start_time = time.time()
normal_result = normal_sum(large_array)
normal_time = time.time() - start_time

start_time = time.time()
numba_result = numba_sum(large_array)
numba_time = time.time() - start_time

print(f"普通函数时间: {normal_time:.4f}秒")
print(f"Numba函数时间: {numba_time:.4f}秒")
print(f"加速比: {normal_time/numba_time:.2f}倍")
